MODULE exit INPUT.

  CASE ok_code.
    WHEN 'CANC'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
   ENDCASE.

ENDMODULE.

MODULE user_command_0100 INPUT.

  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'FC1' OR 'FC2' OR 'FC3' OR 'FC4' OR 'FC5'.
      " Next Screen(다음화면)에 내가 지금 누른 버튼(탭)이 눌리도록 activetab 으로 설정하고,
      PERFORM set_category_data.
      my_tabstrip-activetab = ok_code.  " 이 activetab의 값에 의해 보여질 화면번호가 결정
  ENDCASE.

ENDMODULE.
