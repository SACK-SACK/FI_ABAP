FUNCTION zca_fi_post_recon_ven.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(I_BUKRS) TYPE  ZCA_BUKRS
*"     REFERENCE(I_BELNR) TYPE  ZCA_BELNR
*"     REFERENCE(I_GJAHR) TYPE  ZCA_GJAHR
*"     REFERENCE(I_ITNUM) TYPE  ZCA_ITNUM
*"----------------------------------------------------------------------

*       Global data declarations

  " 넘어온 실제 전표 아이템에 대해서 벤더 원장에 기록한다.
  " 보조원장에 기록될 내용만 기록한다.

  DATA ls_header TYPE zca_fit040. " 전표 헤더
  DATA ls_item TYPE zca_fit050.   " 전표 아이템
  DATA ls_fit100 TYPE zca_fit100. " 보조원장

  " 해당하는 전표 헤더를 가지고 온다.
  SELECT SINGLE * FROM zca_fit040
    INTO ls_header
    WHERE bukrs = i_bukrs
      AND belnr = i_belnr
      AND gjahr = i_gjahr.

  " 해당 하는 아이템을 가지고 온다.
  SELECT SINGLE * FROM zca_fit050
     INTO ls_item
      WHERE bukrs = i_bukrs
      AND belnr = i_belnr
      AND gjahr = i_gjahr
      AND itnum = i_itnum.

  IF sy-subrc EQ 0.

    MOVE-CORRESPONDING ls_item TO ls_fit100.
    ls_fit100-eatax =  ls_fit100-dmbtr * ( '0.1' / '1.1' ). " 부가세
    ls_fit100-lifnr = ls_item-bpcode. " 벤더 코드
    ls_fit100-blart = ls_header-blart.
    ls_fit100-bldat = ls_header-bldat.
    ls_fit100-budat = ls_header-budat.
    ls_fit100-xblnr = ls_header-xblnr.

    " 생성 정보
    ls_fit100-ernam       = sy-uname.
    ls_fit100-erdat       = sy-datum.
    ls_fit100-erzet       = sy-uzeit.

    INSERT zca_fit100 FROM ls_fit100.
  ENDIF.




ENDFUNCTION.
