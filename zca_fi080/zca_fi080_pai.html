MODULE user_command_0100 INPUT.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.

    WHEN 'F010'.
      CALL TRANSACTION 'ZCAFI010'.

    WHEN 'F090'.
      CALL TRANSACTION 'ZCAFI090'.

    WHEN OTHERS.
      CALL METHOD cl_gui_cfw=>dispatch.
  ENDCASE.
  CALL METHOD cl_gui_cfw=>flush.
ENDMODULE.

MODULE exit INPUT.

    WHEN 'EXIT'.

  ENDCASE.
