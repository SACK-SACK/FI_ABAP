FUNCTION zca_fi_bankcode.
*"----------------------------------------------------------------------
*"*"Local Interface:
*"  IMPORTING
*"     REFERENCE(IV_BANKCODE) TYPE  ZCA_BANKCODE
*"----------------------------------------------------------------------

*       Global data declarations

  CLEAR zca_fit080.
  CLEAR screen0170.

  SELECT SINGLE * FROM zca_fit080 INTO CORRESPONDING FIELDS OF zca_fit080
    WHERE bankcode EQ iv_bankcode.

  SELECT SINGLE zltxt FROM zca_tvzbr INTO screen0170-zltxt
    WHERE zlsch EQ zca_fit080-zlsch.

  CALL SCREEN 0170 STARTING AT 5 5.

ENDFUNCTION.
